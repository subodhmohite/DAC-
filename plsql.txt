delimiter //
create procedure updateemp(eid int,esal float(9,2),ejob varchar(20))
begin
update emp
set sal=esal,job=ejob
where empno=eid;
end//
delimiter ;

delimiter $$
create procedure fijob(eid int,out esal float(9,2),out ecomm float(9,2))
begin
select sal,comm into esal,ecomm
from emp
where empno=eid;
end $$

end ..
